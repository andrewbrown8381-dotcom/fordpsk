        function addActivity(action, type = 'info') {
            const activity = {
                id: adminData.activity.length + 1,
                action: action,
                type: type,
                timestamp: new Date().toISOString(),
                date: new Date().toLocaleDateString('ru-RU'),
                time: new Date().toLocaleTimeString('ru-RU', { hour: '2-digit', minute: '2-digit' })
            };
            
            adminData.activity.unshift(activity);
            
            // Сохраняем
            localStorage.setItem('adminActivity', JSON.stringify(adminData.activity));
            
            // Обновляем отображение если нужно
            if (currentAdminTab === 'dashboard') {
                loadRecentActivity();
            }
            if (currentAdminTab === 'profile') {
                loadMyActivity();
            }
        }
        
        function loadAdminStats() {
            const masters = Zapis5MinData.getMasters();
            const bookings = Zapis5MinData.getBookings();
            const users = Zapis5MinData.getUsers();
            
            // Расчет статистики
            const totalMasters = masters.length;
            const totalBookings = bookings.length;
            
            // Рассчитываем доход
            let totalRevenue = 0;
            bookings.forEach(booking => {
                if (booking.status === 'confirmed') {
                    totalRevenue += booking.price || 0;
                }
            });
            
            // Новые клиенты (за последние 30 дней)
            const monthAgo = new Date();
            monthAgo.setDate(monthAgo.getDate() - 30);
            const newClients = users.filter(u => 
                u.role === 'client' && 
                new Date(u.created_at) > monthAgo
            ).length;
            
            // Обновляем статистику
            document.getElementById('totalMasters').textContent = totalMasters;
            document.getElementById('totalBookings').textContent = totalBookings;
            document.getElementById('totalRevenue').textContent = totalRevenue.toLocaleString() + ' ₽';
            document.getElementById('newClients').textContent = newClients;
        }

            let password = '';

        function createMasterAccount() {
            // Собираем данные из формы
            const firstName = document.getElementById('masterFirstName').value.trim();
            const lastName = document.getElementById('masterLastName').value.trim();
            const email = document.getElementById('masterEmail').value.trim();
            const phone = document.getElementById('masterPhone').value.trim();
            const city = document.getElementById('masterCity').value;
            const experience = parseInt(document.getElementById('masterExperience').value) || 0;
            const password = document.getElementById('masterPassword').value;
            const description = document.getElementById('masterDescription').value.trim();
            
            // Проверяем обязательные поля
            if (!firstName || !lastName || !email || !phone || !password) {
                alert('Заполните все обязательные поля: имя, фамилия, email, телефон и пароль');
                return;
            }
            
            // Проверяем email
            if (!isValidEmail(email)) {
                alert('Введите корректный email адрес');
                return;
            }
            
            // Проверяем, не занят ли email
            const users = Zapis5MinData.getUsers();
            if (users.find(u => u.email === email)) {
                alert('Этот email уже зарегистрирован в системе');
                return;
            }
            
            // Собираем специализации
            const specializations = [];
            document.querySelectorAll('input[name="specialization"]:checked').forEach(cb => {
                specializations.push(cb.value);
            });
            
            if (specializations.length === 0) {
                alert('Выберите хотя бы одну специализацию');
                return;
            }
            
            // Генерируем ID для мастера и пользователя
            const masters = Zapis5MinData.getMasters();
            const masterId = masters.length > 0 ? Math.max(...masters.map(m => m.id)) + 1 : 2;
            const userId = users.length > 0 ? Math.max(...users.map(u => u.id)) + 1 : 3;
            
            // Цвет для аватара
            const avatarColors = ['#FF6B8B', '#4CAF50', '#2196F3', '#FF9800', '#9C27B0'];
            const avatarColor = avatarColors[masterId % avatarColors.length];
            
            // Создаем аккаунт пользователя для мастера
            const masterUser = {
                id: userId,
                email: email,
                password: password,
                name: `${firstName} ${lastName}`,
                phone: phone,
                role: 'master',
                is_verified: true,
                created_at: new Date().toISOString(),
                master_id: masterId // Связываем с профилем мастера
            };
            
            // Создаем профиль мастера
            const masterProfile = {
                id: masterId,
                user_id: userId,
                name: `${firstName} ${lastName}`,
                age: 25 + (masterId % 20), // Примерный возраст
                city: city,
                experience: experience,
                rating: 4.5 + (Math.random() * 0.5), // Рейтинг от 4.5 до 5.0
                reviews_count: 0,
                avatar_color: avatarColor,
                telegram: `@${firstName.toLowerCase()}_${lastName.toLowerCase()}`,
                phone: phone,
                specialties: specializations,
                description: description || 'Профессиональный мастер',
                services: createMasterServices(specializations),
                work_schedule: createDefaultWorkSchedule()
            };
            
            // Сохраняем пользователя
            users.push(masterUser);
            Zapis5MinData.updateUsers(users);
            
            // Сохраняем мастера
            masters.push(masterProfile);
            Zapis5MinData.updateMasters(masters);
            
            // Добавляем активность
            addActivity(`Создан аккаунт мастера: ${masterProfile.name}`, 'master');
            
            // Показываем данные для входа
            const loginInfo = `
                ✅ Аккаунт мастера создан успешно!
                
                📧 Email: ${email}
                🔐 Пароль: ${password}
                
                Сохраните эти данные для входа мастера в систему.
                Мастер сможет войти по ссылке: ${window.location.origin}/auth.html
            `;
            
            alert(loginInfo);
            
            // Сбрасываем форму
            document.getElementById('masterFirstName').value = '';
            document.getElementById('masterLastName').value = '';
            document.getElementById('masterEmail').value = '';
            document.getElementById('masterPhone').value = '';
            document.getElementById('masterDescription').value = '';
            
            // Скрываем форму
            hideAddMasterForm();
            
            // Обновляем список мастеров
            loadAllMasters();
            
            // Обновляем статистику
            loadAdminStats();
        }

        function deleteMaster(masterId) {
            if (!confirm('Вы уверены, что хотите удалить мастера? Все связанные записи также будут удалены.')) {
                return;
            }
            
            const masters = Zapis5MinData.getMasters();
            const users = Zapis5MinData.getUsers();
            
            const masterIndex = masters.findIndex(m => m.id === masterId);
            if (masterIndex !== -1) {
                const master = masters[masterIndex];
                
                // Удаляем пользователя мастера
                const userIndex = users.findIndex(u => u.master_id === masterId);
                if (userIndex !== -1) {
                    users.splice(userIndex, 1);
                    Zapis5MinData.updateUsers(users);
                }
                
                // Удаляем мастера
                masters.splice(masterIndex, 1);
                Zapis5MinData.updateMasters(masters);
                
                // Добавляем активность
                addActivity(`Удален мастер: ${master.name}`, 'master');
                
                // Обновляем отображение
                loadAllMasters();
                loadAdminStats();
                
                alert('Мастер успешно удален');
            }
        }

        function getInitials(name) {

        function getStarsHtml(rating) {
            let stars = '';
            for (let i = 1; i <= 5; i++) {
                if (i <= Math.floor(rating)) {
                    stars += '<i class="fas fa-star"></i>';
                } else if (i === Math.ceil(rating) && rating % 1 >= 0.5) {
                    stars += '<i class="fas fa-star-half-alt"></i>';
                } else {
                    stars += '<i class="far fa-star"></i>';
                }
            }
            return stars;
        }
