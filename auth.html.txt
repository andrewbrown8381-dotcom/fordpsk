        function handleRegistration() {
            clearErrors();
            
            registrationData = {
                name: document.getElementById('register-name').value.trim(),
                email: document.getElementById('register-email').value.trim(),
                phone: document.getElementById('register-phone').value.replace(/\D/g, ''),
                password: document.getElementById('register-password').value,
                confirm: document.getElementById('register-confirm').value,

            };
            
            // –í–∞–ª–∏–¥–∞—Ü–∏—è
            let isValid = true;
            
            if (!registrationData.name || registrationData.name.split(' ').filter(w => w.length > 0).length < 2) {
                showError('register-name', '–í–≤–µ–¥–∏—Ç–µ –∏–º—è –∏ —Ñ–∞–º–∏–ª–∏—é');
                isValid = false;
            }
            
            if (!registrationData.email) {
                showError('register-email', '–í–≤–µ–¥–∏—Ç–µ email');
                isValid = false;
            } else if (!isValidEmail(registrationData.email)) {
                showError('register-email', '–í–≤–µ–¥–∏—Ç–µ –∫–æ—Ä—Ä–µ–∫—Ç–Ω—ã–π email');
                isValid = false;
            }

                showError('register-phone', '–í–≤–µ–¥–∏—Ç–µ –Ω–æ–º–µ—Ä —Ç–µ–ª–µ—Ñ–æ–Ω–∞');
                isValid = false;

                showError('register-phone', '–í–≤–µ–¥–∏—Ç–µ 10 —Ü–∏—Ñ—Ä –Ω–æ–º–µ—Ä–∞');
                isValid = false;
            }
            
            if (!registrationData.password) {
                showError('register-password', '–í–≤–µ–¥–∏—Ç–µ –ø–∞—Ä–æ–ª—å');
                isValid = false;
            } else if (registrationData.password.length < 8) {
                showError('register-password', '–ú–∏–Ω–∏–º—É–º 8 —Å–∏–º–≤–æ–ª–æ–≤');
                isValid = false;
            }
            
            if (!registrationData.confirm) {

            const users = Zapis5MinData.getUsers();

                showError('register-email', '–≠—Ç–æ—Ç email —É–∂–µ –∑–∞—Ä–µ–≥–∏—Å—Ç—Ä–∏—Ä–æ–≤–∞–Ω');

        function verifySMS() {
            const smsCode = document.getElementById('sms-code-single').value.trim();
            const errorEl = document.getElementById('sms-error');
            
            if (!smsCode || smsCode.length !== 4) {
                errorEl.textContent = '–í–≤–µ–¥–∏—Ç–µ 4-–∑–Ω–∞—á–Ω—ã–π –∫–æ–¥';
                errorEl.classList.add('show');
                return;
            }
            
            // –î–ª—è –¥–µ–º–æ –ø—Ä–∏–Ω–∏–º–∞–µ–º 1234 –∏–ª–∏ 0000
            if (smsCode !== '1234' && smsCode !== '0000') {
                errorEl.textContent = '–ù–µ–≤–µ—Ä–Ω—ã–π –∫–æ–¥';
                errorEl.classList.add('show');
                return;
            }
            
            // –°–æ–∑–¥–∞–µ–º –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—è
            const users = Zapis5MinData.getUsers();
            const newUserId = users.length > 0 ? Math.max(...users.map(u => u.id)) + 1 : 3;
            
            const newUser = {
                id: newUserId,
                email: registrationData.email,
                password: registrationData.password,
                name: registrationData.name,
                phone: '+7' + registrationData.phone,
                role: 'client',
                is_verified: true,
                created_at: new Date().toISOString()
            };
            
            users.push(newUser);
            Zapis5MinData.updateUsers(users);
            
            // –ê–≤—Ç–æ–º–∞—Ç–∏—á–µ—Å–∫–∏ –≤—Ö–æ–¥–∏–º
            localStorage.setItem('current_user', JSON.stringify(newUser));
            
            alert('–†–µ–≥–∏—Å—Ç—Ä–∞—Ü–∏—è —É—Å–ø–µ—à–Ω–æ –∑–∞–≤–µ—Ä—à–µ–Ω–∞!');
            
            setTimeout(() => {
                window.location.href = 'index.html';
            }, 1500);
        }

        function resendSMS() {
            clearSMSTimer();
            startSMSTimer();
            
            // –î–ª—è –¥–µ–º–æ
            document.getElementById('sms-code-single').value = '1234';
        }

        function startSMSTimer() {
            countdownSeconds = 60;
            const resendBtn = document.getElementById('resend-btn');
            const timerDisplay = document.getElementById('countdown');
            
            if (resendBtn) resendBtn.disabled = true;
            if (timerDisplay) timerDisplay.textContent = countdownSeconds;
            
            smsTimer = setInterval(() => {
                countdownSeconds--;
                if (timerDisplay) timerDisplay.textContent = countdownSeconds;
                
                if (countdownSeconds <= 0) {
                    clearSMSTimer();
                    if (resendBtn) {
                        resendBtn.disabled = false;
                        document.getElementById('resend-timer').style.display = 'none';
                    }
                }
            }, 1000);
        }

        function clearSMSTimer() {
            if (smsTimer) {
                clearInterval(smsTimer);
                smsTimer = null;
            }
        }

        function isValidEmail(email) {
            return /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email);
        }
