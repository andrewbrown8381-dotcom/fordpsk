        // Логика универсальной страницы мастера
        
        document.addEventListener('DOMContentLoaded', function() {
            console.log('Страница мастера загружена');
            
            // Получаем ID мастера из URL
            const urlParams = new URLSearchParams(window.location.search);
            const masterId = urlParams.get('id') || sessionStorage.getItem('selectedMasterId');
            
            console.log('ID мастера:', masterId);
            
            if (!masterId) {
                showError('Мастер не выбран');
                return;
            }
            
            // Загружаем данные мастера
            loadMasterProfile(masterId);
        });

        function loadMasterProfile(masterId) {
            console.log('Загружаем профиль мастера ID:', masterId);
            
            // Ищем мастера в базе данных
            const masters = Zapis5MinData.getMasters();
            const master = masters.find(m => m.id === parseInt(masterId));
            
            if (!master) {
                showError('Мастер не найден');
                return;
            }
            
            console.log('Найден мастер:', master.name);
            
            // Отрисовываем профиль
            renderMasterProfile(master);
        }

        function renderMasterProfile(master) {
            const profileContainer = document.getElementById('profileContainer');
            
            // Получаем инициалы для аватара
            const initials = getInitials(master.name);
            
            // Формируем список специализаций
            const specialties = master.specialties ? master.specialties.map(cat => {
                switch(cat) {
                    case 'MANICURE_PEDICURE': return 'Маникюр';
                    case 'MEN_HAIRCUT': return 'Мужская стрижка';
                    case 'WOMEN_HAIRCUT': return 'Женская стрижка';
                    case 'EYEBROWS': return 'Брови';
                    case 'EYELASHES': return 'Ресницы';
                    default: return cat;
                }
            }).join(', ') : '';
            
            // Создаем HTML профиля
            profileContainer.innerHTML = `
                <div class="profile-content">
                    <div class="profile-avatar">
                        <div class="avatar-large" style="background: ${master.avatar_color || '#8A2BE2'}">
                            ${initials}
                        </div>
                    </div>
                    
                    <div class="profile-info">
                        <h1 class="profile-name">${master.name}</h1>
                        
                        <div class="profile-details">
                            <div class="detail-item">
                                <i class="fas fa-user"></i>
                                <span>Возраст: ${master.age || 'не указан'} лет</span>
                            </div>
                            <div class="detail-item">
                                <i class="fas fa-map-marker-alt"></i>
                                <span>Город: ${master.city || 'не указан'}</span>
                            </div>
                            <div class="detail-item">
                                <i class="fas fa-briefcase"></i>
                                <span>Опыт: ${master.experience || 0} лет</span>
                            </div>
                            <div class="detail-item">
                                <i class="fas fa-star"></i>
                                <span>Рейтинг: ${master.rating || 0} (${master.reviews_count || 0} отзывов)</span>
                            </div>
                            <div class="detail-item">
                                <i class="fas fa-comment-alt"></i>
                                <span>${master.description || 'Профессиональный мастер'}</span>
                            </div>
                            <div class="detail-item">
                                <i class="fas fa-spa"></i>
                                <span>Специализация: ${specialties}</span>
                            </div>
                        </div>
                        
                        <div class="telegram-btn-container">
                            <button class="telegram-btn" onclick="openTelegram('${master.telegram || ''}')">
                                <i class="fab fa-telegram"></i> НАПИСАТЬ В ТЕЛЕГРАМ
                            </button>
                            <p class="telegram-note">Написать мастеру для уточнения деталей</p>
                        </div>
                    </div>
                </div>
                
                <!-- Вкладки -->
                <div class="tabs-container">
                    <div class="tabs-header">
                        <button class="tab-btn active" onclick="switchTab('services', ${master.id})">УСЛУГИ И ЦЕНЫ</button>
                        <button class="tab-btn" onclick="switchTab('booking', ${master.id})">ЗАПИСЬ</button>
                    </div>
                    
                    <!-- Вкладка услуг -->
                    <div id="tab-services" class="tab-content active">
                        ${renderServicesTable(master)}
                    </div>
                    
                    <!-- Вкладка записи -->
                    <div id="tab-booking" class="tab-content">
                        ${renderBookingForm(master)}
                    </div>
                </div>
            `;
            
            // Инициализируем первую вкладку
            switchTab('services', master.id);
        }

        function renderServicesTable(master) {
            if (!master.services || master.services.length === 0) {
                return `
                    <div style="text-align: center; padding: 40px; color: var(--text-gray);">
                        <i class="fas fa-spa" style="font-size: 48px; margin-bottom: 20px;"></i>
                        <p>У мастера пока нет услуг</p>
                    </div>
                `;
            }
            
            let tableHTML = `
                <table class="services-table">
                    <thead>
                        <tr>
                            <th>УСЛУГА</th>
                            <th>ЦЕНА</th>
                            <th>ДЛИТЕЛЬНОСТЬ</th>
                            <th>ВЫБРАТЬ</th>
                        </tr>
                    </thead>
                    <tbody>
            `;
            
            master.services.forEach(service => {
                tableHTML += `
                    <tr>
                        <td>
                            <div style="color: var(--text-light); font-weight: 500;">${service.name}</div>
                            ${service.description ? `<div style="color: var(--text-gray); font-size: 14px; margin-top: 5px;">${service.description}</div>` : ''}
                        </td>
                        <td><strong>${service.price.toLocaleString()} ₽</strong></td>
                        <td>${service.duration || 60} мин</td>
                        <td>
                            <button class="service-select-btn" onclick="selectService(${master.id}, ${service.id})">
                                Выбрать
                            </button>
                        </td>
                    </tr>
                `;
            });
            
            tableHTML += `
                    </tbody>
                </table>
            `;
            
            return tableHTML;
        }

        function renderBookingForm(master) {
            if (!master.services || master.services.length === 0) {
                return `
                    <div style="text-align: center; padding: 40px; color: var(--text-gray);">
                        <i class="fas fa-calendar-times" style="font-size: 48px; margin-bottom: 20px;"></i>
                        <p>У мастера пока нет услуг для записи</p>
                    </div>
                `;
            }
            
            let optionsHTML = '';
            master.services.forEach(service => {
                optionsHTML += `<option value="${service.id}">${service.name} - ${service.price.toLocaleString()} ₽ (${service.duration || 60} мин)</option>`;
            });
            
            return `
                <div class="booking-form">
                    <div class="form-group">
                        <label class="form-label">Выберите услугу:</label>
                        <select class="form-select" id="bookingServiceSelect">
                            ${optionsHTML}
                        </select>
                    </div>
                    
                    <button class="booking-btn" onclick="bookService(${master.id})">
                        ЗАПИСАТЬСЯ НА УСЛУГУ
                    </button>
                </div>
            `;
        }

        function switchTab(tabName, masterId) {
            // Скрываем все вкладки
            document.querySelectorAll('.tab-content').forEach(tab => {
                tab.style.display = 'none';
                tab.classList.remove('active');
            });
            
            // Убираем активный класс со всех кнопок
            document.querySelectorAll('.tab-btn').forEach(btn => {
                btn.classList.remove('active');
            });
            
            // Показываем нужную вкладку
            document.getElementById(`tab-${tabName}`).style.display = 'block';
            document.getElementById(`tab-${tabName}`).classList.add('active');
            
            // Активируем кнопку
            document.querySelectorAll('.tab-btn').forEach(btn => {
                if (btn.textContent.includes(tabName === 'services' ? 'УСЛУГИ' : 'ЗАПИСЬ')) {
                    btn.classList.add('active');
                }
            });
        }

        function selectService(masterId, serviceId) {
            console.log('Выбрана услуга:', serviceId, 'у мастера:', masterId);
            
            // Сохраняем выбранные данные
            sessionStorage.setItem('selectedMasterId', masterId);
            sessionStorage.setItem('selectedServiceId', serviceId);
            
            // Переходим на вкладку записи
            switchTab('booking', masterId);
            
            // Выбираем эту услугу в выпадающем списке
            const select = document.getElementById('bookingServiceSelect');
            if (select) {
                select.value = serviceId;
            }
        }

        function bookService(masterId) {
            // Проверяем авторизацию
            const currentUser = JSON.parse(localStorage.getItem('current_user'));
            if (!currentUser) {
                alert('Для записи необходимо войти в систему');
                window.location.href = 'auth.html?redirect=booking';
                return;
            }
            
            const serviceSelect = document.getElementById('bookingServiceSelect');
            if (!serviceSelect) {
                alert('Сначала выберите услугу');
                return;
            }
            
            const serviceId = serviceSelect.value;
            
            // Сохраняем данные для страницы записи
            const bookingData = {
                masterId: masterId,
                serviceId: parseInt(serviceId)
            };
            
            sessionStorage.setItem('bookingData', JSON.stringify(bookingData));
            
            // Переходим на страницу записи
            window.location.href = 'booking.html';
        }

        function openTelegram(telegramUsername) {
            if (!telegramUsername || telegramUsername === 'undefined') {
                alert('У мастера не указан Telegram');
                return;
            }
            
            // Очищаем username от @ если есть
            const cleanUsername = telegramUsername.replace('@', '');
            const telegramUrl = `https://t.me/${cleanUsername}`;
            
            window.open(telegramUrl, '_blank');
        }

        function showError(message) {
            const profileContainer = document.getElementById('profileContainer');
            profileContainer.innerHTML = `
                <div class="error-message">
                    <i class="fas fa-exclamation-triangle"></i>
                    <h3>${message}</h3>
                    <p>Попробуйте выбрать мастера снова</p>
                    <button class="back-btn" onclick="window.location.href = 'masters.html'" style="margin-top: 20px;">
                        <i class="fas fa-arrow-left"></i> Вернуться к выбору мастеров
                    </button>
                </div>
            `;
        }

        // Вспомогательные функции
        function getInitials(name) {
            if (!name) return 'МК';
            return name.split(' ').map(word => word[0]).join('').toUpperCase().slice(0, 2);
        }
