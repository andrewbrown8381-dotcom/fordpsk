    <script>
        // Логика страницы подтверждения
        document.addEventListener('DOMContentLoaded', function() {
            const bookingId = sessionStorage.getItem('confirmedBookingId');
            if (!bookingId) {
                showMessage('Данные записи не найдены', 'error');
                setTimeout(() => window.location.href = 'index.html', 2000);
                return;
            }
            
            const bookings = BeautyProData.getBookings();
            const booking = bookings.find(b => b.id === parseInt(bookingId));
            
            if (!booking) {
                showMessage('Запись не найдена', 'error');
                setTimeout(() => window.location.href = 'index.html', 2000);
                return;
            }
            
            // Получаем дополнительную информацию
            const master = BeautyProData.getMasterById(booking.master_id);
            const service = BeautyProData.getServiceById(booking.service_id);
            const client = checkAuth();
            
            if (!master || !service) {
                showMessage('Ошибка загрузки данных', 'error');
                return;
            }
            
            // Форматируем дату
            const dateFormatter = new Intl.DateTimeFormat('ru-RU', {
                day: 'numeric',
                month: 'long',
                year: 'numeric'
            });
            
            const bookingDate = new Date(booking.date);
            const dateString = dateFormatter.format(bookingDate);
            
            // Заполняем детали
            const bookingDetails = document.getElementById('bookingDetails');
            bookingDetails.innerHTML = `
                <div class="detail-item">
                    <span class="detail-label"><i class="fas fa-spa"></i> Услуга:</span>
                    <span class="detail-value">${service.name}</span>
                </div>
                <div class="detail-item">
                    <span class="detail-label"><i class="fas fa-user-tie"></i> Мастер:</span>
                    <span class="detail-value">${master.name}</span>
                </div>
                <div class="detail-item">
                    <span class="detail-label"><i class="fas fa-calendar-alt"></i> Дата и время:</span>
                    <span class="detail-value">${dateString}, ${booking.time}</span>
                </div>
                <div class="detail-item">
                    <span class="detail-label"><i class="fas fa-map-marker-alt"></i> Адрес:</span>
                    <span class="detail-value">ул. Красивая, д. 1, каб. 5</span>
                </div>
                <div class="detail-item">
                    <span class="detail-label"><i class="fas fa-phone"></i> Телефон мастера:</span>
                    <span class="detail-value">${master.phone}</span>
                </div>
                <div class="detail-item">
                    <span class="detail-label"><i class="fas fa-credit-card"></i> Предоплата:</span>
                    <span class="detail-value">${booking.prepayment.toLocaleString()} ₽ (оплачено)</span>
                </div>
                <div class="detail-total">
                    <span><i class="fas fa-wallet"></i> К оплате на месте:</span>
                    <span>${(booking.price - booking.prepayment).toLocaleString()} ₽</span>
                </div>
            `;
            
            // Сохраняем данные для календаря
            window.bookingForCalendar = {
                title: `${service.name} - ${master.name}`,
                description: `Запись к мастеру красоты\nМастер: ${master.name}\nУслуга: ${service.name}\nТелефон: ${master.phone}`,
                location: 'ул. Красивая, д. 1, каб. 5',
                startDate: `${booking.date}T${booking.time}`,
                endDate: getEndTime(booking.date, booking.time, service.duration || 60)
            };
        });
        
        function getEndTime(date, startTime, durationMinutes) {
            const [hours, minutes] = startTime.split(':').map(Number);
            const startDateTime = new Date(`${date}T${startTime}`);
            const endDateTime = new Date(startDateTime.getTime() + durationMinutes * 60000);
            
            return endDateTime.toISOString().slice(0, 16).replace('T', ' ');
        }
        
        function addToCalendar() {
            if (!window.bookingForCalendar) return;
            
            const { title, description, location, startDate, endDate } = window.bookingForCalendar;
            
            // Создаем ссылку для Google Calendar
            const googleCalendarUrl = `https://calendar.google.com/calendar/render?action=TEMPLATE&text=${encodeURIComponent(title)}&details=${encodeURIComponent(description)}&location=${encodeURIComponent(location)}&dates=${startDate.replace(/[-:]/g, '')}/${endDate.replace(/[-:]/g, '')}`;
            
            // Открываем в новом окне
            window.open(googleCalendarUrl, '_blank');
            
            showMessage('Ссылка на добавление в календарь открыта', 'success');
        }
        
        function sendToEmail() {
            const currentUser = checkAuth();
            if (!currentUser) return;
            
            // Для демо просто показываем сообщение
            showMessage('На вашу почту отправлено письмо с деталями записи', 'success');
            
            // В реальном приложении здесь был бы AJAX запрос к серверу
            console.log('Отправка email на адрес:', currentUser.email);
        }
