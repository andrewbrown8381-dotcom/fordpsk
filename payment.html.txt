    <script>
        // Логика страницы оплаты
        let paymentTimer;
        let remainingTime = 600; // 10 минут в секундах
        let currentBooking = null;

        document.addEventListener('DOMContentLoaded', function() {
            // Проверяем авторизацию
            const currentUser = JSON.parse(localStorage.getItem('current_user'));
            if (!currentUser) {
                alert('Необходима авторизация');
                window.location.href = 'auth.html';
                return;
            }
            
            // Загружаем данные бронирования
            const savedBooking = sessionStorage.getItem('currentBooking');
            if (!savedBooking) {
                alert('Данные бронирования не найдены');
                window.location.href = 'booking.html';
                return;
            }
            
            currentBooking = JSON.parse(savedBooking);
            
            // Заполняем информацию
            document.getElementById('paymentAmount').textContent = 
                '1 250 ₽'; // Для демо
            
            document.getElementById('bookingCode').textContent = 
                currentBooking.booking_code || '#Z-' + new Date().getFullYear() + '-001234';
            
            // Запускаем таймер
            startPaymentTimer();
            
            // Настраиваем валидацию чекбокса
            const paidCheckbox = document.getElementById('paidCheckbox');
            const confirmBtn = document.getElementById('confirmPaymentBtn');
            
            paidCheckbox.addEventListener('change', function() {
                confirmBtn.disabled = !this.checked;
            });
        });

        function startPaymentTimer() {
            updateTimerDisplay();
            
            paymentTimer = setInterval(() => {
                remainingTime--;
                updateTimerDisplay();
                
                if (remainingTime <= 0) {
                    clearInterval(paymentTimer);
                    showPaymentExpired();
                }
            }, 1000);
        }

        function updateTimerDisplay() {
            const minutes = Math.floor(remainingTime / 60);
            const seconds = remainingTime % 60;
            
            document.getElementById('paymentTimer').textContent = 
                `${minutes.toString().padStart(2, '0')}:${seconds.toString().padStart(2, '0')}`;
            
            if (remainingTime <= 60) {
                document.getElementById('paymentTimer').style.color = '#ff4757';
            }
        }

        function showPaymentExpired() {
            alert('Время на оплату истекло. Бронирование отменено.');
            window.location.href = 'booking.html';
        }

        function moveCardDigit(current) {
            const currentInput = document.getElementById(`cardDigit${current}`);
            const nextInput = document.getElementById(`cardDigit${current + 1}`);
            
            if (currentInput.value && nextInput) {
                nextInput.focus();
            }
        }

        function confirmPayment() {
            const paidCheckbox = document.getElementById('paidCheckbox');
            if (!paidCheckbox.checked) {
                alert('Подтвердите, что вы оплатили предоплату');
                return;
            }
            
            let cardDigits = '';
            for (let i = 1; i <= 4; i++) {
                const digit = document.getElementById(`cardDigit${i}`).value;
                if (!digit || isNaN(digit)) {
                    alert('Введите последние 4 цифры карты');
                    document.getElementById(`cardDigit${i}`).focus();
                    return;
                }
                cardDigits += digit;
            }
            
            clearInterval(paymentTimer);
            
            alert('Оплата подтверждена! Создаем запись...');
            
            setTimeout(() => {
                sessionStorage.setItem('confirmedBookingId', '1');
                window.location.href = 'booking-success.html';
            }, 1500);
        }

        function goBack() {
            window.history.back();
        }
    </script>
